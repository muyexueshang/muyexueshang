/*
归并排序:MERGE_SORT
时间复杂度：O(nlogn)
空间复杂度:O(n)
*/
void merge(int array[],int array_temp[],int head,int middle,int array_length)
{
	int k=head,j=middle+1;
	while(head<=middle&&j<=array_length)
	{
		array[head]<array[j]?array_temp[k++]=array[head++]:array_temp[k++]=array[j++];
	}
	while(head<=middle)
	{
		array_temp[k++]=array[head++];
	}
	while(j<=array_length)
	{
		array_temp[k++]=array[j++];
	}
}
void merge_sort(int array[],int array_temp[],int array_length)
{
	int length=1,*p=array,*temp;
	while(length<array_length)
	{
		int old_length=length,i=0;
		length*=2;
		while(i+length<array_length)
		{
			merge(p,array_temp,i,i+old_length-1,i+length-1);
			i+=length;
		}
		if(i+old_length<array_length)
		{
			merge(p,array_temp,i,i+old_length-1,array_length-1);
		}
		temp=p;
		p=array_temp;
		array_temp=temp;
	}
}
